// ─────────────────────────────────────────────────────────────────────
// DATA MODEL
// Each "task" belongs to exactly one (feature, person) — stored flat.
// person = 'pre' | 'huy' | 'tuan' | 'bang' | 'hoa'
// Each task occupies its own <tr>; the feature cell rowSpans all tasks.
// ─────────────────────────────────────────────────────────────────────

const PEOPLE  = ['huy','tuan','bang','hoa'];
const P_LABEL = { huy:'Huy', tuan:'Tuấn', bang:'Bằng', hoa:'Hòa' };
const P_INIT  = { huy:'HY',  tuan:'TN',   bang:'BG',   hoa:'HÒ'  };
const P_COLOR = { huy:'blue', tuan:'green', bang:'orange', hoa:'purple' };
const ACCENTS = ['blue','green','orange','purple'];
const ADOTS   = { blue:'#1a73e8', green:'#1e8e3e', orange:'#e37400', purple:'#9334e6' };

// Each feature has an array of tasks.
// task = { id, name, person ('pre'|person), done }
let db = {
  features: [
    {
      id:'f1', name:'Refactor\nPS Mode', accent:'blue',
      tasks: [
        { id:'t1', label:'01', name:'lấy file cho design', person:'pre',  done:false },

      ]
    },

        { id:'t3', label:'1.1', name:'Check the feedback and split item', person:'pre', done:false },
      ]
    },
    {
      id:'f3', name:'Custom text', accent:'orange',
      tasks: [
        { id:'t4', label:'2.1', name:'implement',   person:'hoa',  done:true  },
        { id:'t5', label:'2.1', name:'Code review', person:'tuan', done:false },
        { id:'t6', label:'2.2', name:'Code review', person:'bang', done:false },
      ]
    },
    {
      id:'f4', name:'IXL Time spent', accent:'purple',
      tasks: [
        { id:'t7', label:'3.1', name:'investigate',      person:'tuan', done:false },
        { id:'t8', label:'3.2', name:'discuss with huy', person:'huy',  done:false },
        { id:'t9', label:'3.3', name:'implement',        person:'pre',  done:false },
      ]
    }
  ]
};

let uid = 100;
const newId = () => 'x' + (uid++);

// ── RENDER ────────────────────────────────────────────────────────────
function render() {
  renderSummary();
  renderHeader();
  renderBody();
}

function renderHeader() {
  document.getElementById('thead').innerHTML = `<tr>
    <th style="min-width:var(--feature-w);width:var(--feature-w)">
      <div class="th-in">Feature</div>
    </th>
    <th style="min-width:var(--pre-w);width:var(--pre-w)">
      <div class="th-in">
        <svg width="10" height="10" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5" style="vertical-align:middle;margin-right:4px"><polygon points="13 2 3 14 12 14 11 22 21 10 12 10 13 2"/></svg>
        Làm trước
      </div>
    </th>
    ${PEOPLE.map(p => `
      <th style="min-width:var(--col-w);width:var(--col-w)">
        <div class="th-in th-person">
          <div class="av" style="background:var(--${P_COLOR[p]}-bg);color:var(--${P_COLOR[p]})">${P_INIT[p]}</div>
          ${P_LABEL[p]}
        </div>
      </th>`).join('')}
  </tr>`;
}

function renderBody() {
  const tbody = document.getElementById('tbody');
  tbody.innerHTML = '';

  db.features.forEach(feat => {
    const tasks = feat.tasks;
    const rowCount = tasks.length; // each task = 1 row

    tasks.forEach((task, ti) => {
      const isFirst = ti === 0;
      const isLast  = ti === rowCount - 1;
      const cols = ['pre', ...PEOPLE]; // all 5 columns

      const tr = document.createElement('tr');
      tr.className = [
        `ab-${feat.accent}`,
        task.done ? 'done-row' : '',
        isLast ? 'feat-boundary' : '',
        ti % 2 === 1 ? 'task-row-even' : ''
      ].filter(Boolean).join(' ');
      tr.dataset.fid = feat.id;
      tr.dataset.tid = task.id;

      // Feature cell (rowspan entire feature)
      if (isFirst) {
        const td = document.createElement('td');
        td.className = 'feat-td';
        td.rowSpan = rowCount;
        td.innerHTML = `<div class="feat-inner">
          <div class="feat-dot" style="background:${ADOTS[feat.accent]}"></div>
          <span class="feat-name">${feat.name}</span>
          <span class="feat-count">${tasks.length} task${tasks.length!==1?'s':''}</span>
        </div>`;
        tr.appendChild(td);
      }

      // Content columns: pre + each person
      cols.forEach(col => {
        const td = document.createElement('td');
        td.className = `content-cell${col !== 'pre' ? ' person-col' : ''}`;
        td.dataset.fid    = feat.id;
        td.dataset.tid    = task.id;
        td.dataset.col    = col;

        if (task.person === col) {
          // This is the task's owner column — show the card
          const card = makeCard(task, feat.id);
          td.appendChild(card);
        } else {
          // Empty — show add hint on hover
          const hint = document.createElement('div');
          hint.className = 'empty-hint';
          hint.innerHTML = `<svg width="9" height="9" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3"><line x1="12" y1="5" x2="12" y2="19"/><line x1="5" y1="12" x2="19" y2="12"/></svg> thêm`;
          hint.addEventListener('click', () => {
            // Move this task to this column
            moveTaskPerson(feat.id, task.id, col);
          });
          td.appendChild(hint);
        }

        // Drag/drop
        td.addEventListener('dragover', e => { e.preventDefault(); td.classList.add('drag-over'); });
        td.addEventListener('dragleave', () => td.classList.remove('drag-over'));
        td.addEventListener('drop', e => onDrop(e, feat.id, task.id, col));

        tr.appendChild(td);
      });

      tbody.appendChild(tr);
    });

    // "Add task" hint row at bottom of feature
    const addTr = document.createElement('tr');
    addTr.className = `add-row-hint ab-${feat.accent}`;
    const spaceTd = document.createElement('td');
    spaceTd.className = 'feat-td-spacer';
    // don't add feat-td here — just a spacer
    spaceTd.style.cssText = `min-width:var(--feature-w);width:var(--feature-w);border-right:2px solid var(--border);background:white;position:sticky;left:0;z-index:10;`;
    addTr.appendChild(spaceTd);

    const addTd = document.createElement('td');
    addTd.colSpan = 5;
    addTd.style.cssText = 'padding:4px 10px;';
    addTd.innerHTML = `<div class="add-row-btn" onclick="addTaskPrompt('${feat.id}')">
      <svg width="10" height="10" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><line x1="12" y1="5" x2="12" y2="19"/><line x1="5" y1="12" x2="19" y2="12"/></svg>
      Thêm task
    </div>`;
    addTr.appendChild(addTd);
    tbody.appendChild(addTr);
  });
}

function makeCard(task, fid) {
  const card = document.createElement('div');
  card.className = `task-card card-${task.person}${task.done ? ' done' : ''}`;
  card.draggable = true;
  card.dataset.fid  = fid;
  card.dataset.tid  = task.id;
  card.dataset.person = task.person;
  card.title = 'Click → done/undone';
  card.innerHTML = `
    <span class="task-id">#${task.label}</span>
    <span class="task-name">${task.name}</span>
    ${task.done ? '<span class="done-check">✓</span>' : ''}
  `;
  card.addEventListener('click', () => { toggleDoneTask(fid, task.id); });
  card.addEventListener('dragstart', e => {
    drag = { fid, tid: task.id, person: task.person };
    setTimeout(() => card.classList.add('dragging'), 0);
    e.dataTransfer.effectAllowed = 'move';
  });
  card.addEventListener('dragend', () => card.classList.remove('dragging'));
  return card;
}

// ── INTERACTIONS ──────────────────────────────────────────────────────
function toggleDoneTask(fid, tid) {
  const feat = db.features.find(f => f.id === fid);
  const task = feat?.tasks.find(t => t.id === tid);
  if (task) { task.done = !task.done; render(); applyHighlight(); }
}

function moveTaskPerson(fid, tid, newPerson) {
  const feat = db.features.find(f => f.id === fid);
  const task = feat?.tasks.find(t => t.id === tid);
  if (task) { task.person = newPerson; render(); applyHighlight(); }
}

function addTaskPrompt(fid) {
  const feat = db.features.find(f => f.id === fid);
  if (!feat) return;
  const name = prompt(`Task mới cho "${feat.name.replace('\n',' ')}":`);
  if (!name) return;
  const label = prompt('ID (ví dụ: 4.1):') || String(feat.tasks.length + 1);
  const personOptions = ['pre (làm trước)', ...PEOPLE.map(p => P_LABEL[p])];
  const personInput = prompt(`Ai phụ trách?\n0: Làm trước\n${PEOPLE.map((p,i)=>`${i+1}: ${P_LABEL[p]}`).join('\n')}\nNhập số:`);
  const idx = parseInt(personInput);
  const person = idx === 0 ? 'pre' : (PEOPLE[idx - 1] || 'pre');
  feat.tasks.push({ id: newId(), label, name, person, done: false });
  render();
}

function addFeaturePrompt() {
  const name = prompt('Tên feature mới:');
  if (!name) return;
  const accent = ACCENTS[db.features.length % ACCENTS.length];
  db.features.push({ id: newId(), name, accent, tasks: [] });
  render();
}

function toggleDone(cb) {
  document.body.classList.toggle('hide-done', !cb.checked);
}

// ── DRAG / DROP ───────────────────────────────────────────────────────
let drag = null;

function onDrop(e, toFid, toRowTid, toCol) {
  e.preventDefault();
  document.querySelectorAll('.drag-over').forEach(el => el.classList.remove('drag-over'));
  if (!drag) return;

  const fromFeat = db.features.find(f => f.id === drag.fid);
  const fromTask = fromFeat?.tasks.find(t => t.id === drag.tid);
  if (!fromTask) return;

  if (drag.fid === toFid && drag.tid === toRowTid) {
    // Same row — just reassign person column
    fromTask.person = toCol;
  } else {
    // Different row — move task to target feature, change person
    fromFeat.tasks = fromFeat.tasks.filter(t => t.id !== drag.tid);
    const toFeat = db.features.find(f => f.id === toFid);
    // Insert after toRowTid's index
    const insertIdx = toFeat.tasks.findIndex(t => t.id === toRowTid);
    fromTask.person = toCol;
    toFeat.tasks.splice(insertIdx + 1, 0, fromTask);
  }

  drag = null;
  render();
  applyHighlight();
}

// ── SUMMARY / HIGHLIGHT ───────────────────────────────────────────────
let activeHL = null;

function renderSummary() {
  const bar = document.getElementById('summaryBar');
  bar.innerHTML = '';
  PEOPLE.forEach(p => {
    const tasks  = db.features.flatMap(f => f.tasks).filter(t => t.person === p);
    const done   = tasks.filter(t => t.done).length;
    const total  = tasks.length;
    const pill   = document.createElement('div');
    pill.className = `summary-pill pill-${p}`;
    pill.innerHTML = `
      <div class="av" style="background:var(--${P_COLOR[p]}-bg);color:var(--${P_COLOR[p]})">${P_INIT[p]}</div>
      <span class="pill-count">${total}</span>
      <div>
        <div class="pill-name">${P_LABEL[p]}</div>
        <div class="pill-sub">${done} done · ${total - done} active</div>
      </div>`;
    pill.addEventListener('click', () => {
      activeHL = activeHL === p ? null : p;
      applyHighlight();
    });
    bar.appendChild(pill);
  });
}

function applyHighlight() {
  // Sync pill states
  document.querySelectorAll('.summary-pill').forEach(el => {
    const p = [...el.classList].find(c => c.startsWith('pill-'))?.replace('pill-','');
    el.classList.toggle('dimmed', activeHL !== null && activeHL !== p);
  });
  // Fade non-active cards
  document.querySelectorAll('.task-card').forEach(c => {
    c.classList.toggle('faded', activeHL !== null && c.dataset.person !== activeHL);
  });
}

// ── INIT ──────────────────────────────────────────────────────────────
render();
